\begin{lstlisting}[escapechar=!]
int main()
{
	int** ipp;

	int x1 = 5;
	int x2 = 6;
	int x3 = 7;

	int* ip1;
	int* ip2;

	ip1 = &x1;
	ip2 = &x2;
	ipp = &ip1;

	*ipp = ip2;
	*ipp = &x3;
	**ipp = -1;

	return 0;
