\noindent \textbf{Question:} Overload \texttt{operator!=} for the Complex and Matrix classes given below.\\
\begin{lstlisting}
class Complex
{
	private:
	float real;
	float img;
	public:
	// Write code for operator!= here.
	bool operator!= (const Complex& rhs) // const is not needed but it's good programming practice.
	{
		if (real != rhs.real || img != rhs.img)
			return true;
		else
			return false;
	}	
};
class Matrix
{
	private:
	int Size;
	float* Data;
	public:
	// Write code for operator!= here.
	bool operator!= (const Matrix& rhs)
	{
		// If matrix sizes don't match, they're not equal.
		if (Size != rhs.Size)
			return true;
		for (int i=0; i<Size; i++)
		{
			if (Data[i] != rhs.Data[i]) // If any mismatch found return true.
				return true;
		}
		return false; // If we're here then all entries matched.
	}
};
\end{lstlisting}
