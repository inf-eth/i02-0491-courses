\section{Linked List}
\begin{itemize}
\item A node can be added at start of linked list, middle or at end.
\end{itemize}
\begin{lstlisting}[caption={Linked List: Add at Head and Display}]
#include <iostream>
using namespace std;

struct Node
{
	int Data;
	Node* Next;
};

Node* First = NULL;

void AddAtHead(int pData)
{
	Node* NewNode;
	NewNode = new Node;
	NewNode->Data = pData;

	if (First == NULL)
	{
		First = NewNode;
		NewNode->Next = NULL;
	}
	else
	{
		NewNode->Next = First;
		First = NewNode;
	}
}

void Display()
{
	Node* temp;
	temp = First;
	if (temp == NULL)
		return;
	while (temp != NULL)
	{
		cout << temp->Data << " ";
		temp = temp->Next;
	}
	cout << endl;
}

int main()
{
	AddAtHead(3);
	AddAtHead(4);
	AddAtHead(7);
	AddAtHead(-1);
	Display();
	
	return 0;
}
\end{lstlisting}
