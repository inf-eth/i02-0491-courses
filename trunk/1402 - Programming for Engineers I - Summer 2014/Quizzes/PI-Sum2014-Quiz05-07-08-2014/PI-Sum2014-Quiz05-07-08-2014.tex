\noindent\textbf{Question:} Write code for Server/Client authorisation. Client will send password to Server in a loop. If password matches loop will break otherwise Client re--enters password.
\begin{lstlisting}
// Server
int a = 0;
char pass[] = "123";
char temp[20];
while (true)
{
	ServerObj.Receive((void*)temp, 20);
	if (strcmp(temp, pass)==0)
		a = 1;
	ServerObj.Send((void*)&a, sizeof(int));
	if (a==1)
		break;
}
\end{lstlisting}
\begin{lstlisting}
// Client
int a = 0;
char temp[20];
while (true)
{
	cout << "Enter password: ";
	cin.getline(temp, 20);
	ClientObj.Send((void*)temp, strlen(temp)+1);
	ClientObj.Receive((void*)&a, sizeof(int));
	if (a==1)
	{
		cout << "Successful login!" << endl;
		break;
	}
	else
		cout << "Incorrect password!" << endl;
}
\end{lstlisting}
