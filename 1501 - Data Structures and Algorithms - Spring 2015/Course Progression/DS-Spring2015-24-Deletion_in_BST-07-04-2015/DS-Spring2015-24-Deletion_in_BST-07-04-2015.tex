// Delete takes a node pointer (which is root of some tree/subtree), pointer to node's parent and value to be deleted.
// Initially called with Delete(Root, Root, x);
bool Delete(Node* n, Node* p, int x)
	Node* temp = n
	Node* parent
	if n is equal to p
	then parent = NULL
	else parent = p

	while true
		if temp is NULL
			return false
		if x is equal to temp.x
			break
		else if x < temp.x
			parent = temp
			temp = temp.left
		else if x > temp.x
			parent = temp
			temp = temp.right
		else
			return false
	while end

	// 1. Deletion if leaf
	if temp.left and temp.right are both NULL
		if parent is not NULL
			if parent.left is equal to temp
				parent.left = NULL
			else
				parent.right - NULL
		delete temp
		return true

	// 2. Deletion if only one child
	else if temp.left is NULL OR temp.right is NULL
		if parent is not NULL
			if parent.left is equal to temp
				if temp.left is NULL
					parent.left = temp.right
				else
					parent.left = temp.left
			else
				if temp.left is NULL
					parent.right = temp.right
				else
					parent.right = temp.left
		delete temp
		return true

	// 3. If two children
	else if temp.left AND temp.right are not NULL
		min = FindMin(temp.right)
		temp.x = min
		return Delete(temp.right, temp, min)
	else
		return false
\end{lstlisting}
